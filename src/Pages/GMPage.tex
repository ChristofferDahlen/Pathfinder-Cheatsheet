        \begin{Table}{Turn}
            \begin{entry}{Rolling initative}{%
                align=center seam,
                name width=\turnLength,
                page={\raisebox{-2pt}{\stackbox{435,\\GMC 25}}},
            }%
                You usually roll \PerceptionT for initiative.
                \GM might call for other checks if relevant. \hfill
                \D{For ties: enemies go first and players choose order during first round.}\\
                If creatures use \Refer{Avoid Notice} the \StealthT roll can be used for initiative.\hfill
                \D{Creatues have to use \StealthT or become noticed.}
            \end{entry}
            \begin{entry}{Start of turn}{%
                align=center seam,
                name width=\turnLength,%
                page=435,
            }
                - Use\A{f} or your\A{r} with trigger ``Your turn begins''. \hfill
                - Reduce your active effects by 1 round. \hfill
                - Recovery check. \hfill - Other start of turn effects.\\
                The last step you regain your actions\A{a} and your reactions\A{r}. \hfill
                \D{You also gain/lose extra actions here.}
            \end{entry}
            \begin{entry}{Turn}{%
                name width=\turnLength,%
                page=436,
            }
                You use action/reactions/activities until you cannot. \hfill
                \D{If you can't act, you can't use any actions, including reactions and free actions.}
            \end{entry}
            \begin{entry}{End of Turn}{%
                align=center seam,
                name width=\turnLength,%
                page=436,
            }%
                \begin{tabular}{@{}ll}%
                    - Use\A{f} or your\A{r} with a trigger ``Your turn ends'' &%
                    - End effects that last until end of turn. \\%
                    - Take persistent damage, attempt the flat check. &
                    - Resolve anything specified.
                \end{tabular}%}
            \end{entry}
            \begin{entry}{Reactions}{%
                name width=\turnLength,%
                page={15, 436},
            }
                You can use reactions \A{r} outside your turn. \hfill
                \D{\GM determins wheter you can use reactions before your first turn begins, depending on situtation.}
            \end{entry}
            \begin{entry}{Free Actions}{%
                align=center seam,
                name width=\turnLength,%
                page=15,
            }
                You can use however many free actions\A{f} during your turn. \hfill
                \pbox{0.6\linewidth}{If it has a trigger, like a reaction, then it can be triggered as a reaction.\\
                \D{You can only use one free action per trigger.}}
            \end{entry}
        \end{Table}\TableSpace
        \begin{Table}{Falling}[][\pghh{421}]
            \begin{entry}{Falling}{%
                align=center seam,
                name width=\turnLength,}
                If you fall more than \Numb{5} \Feet you take \Type{Bludgeoning} equal to half you fell.\hfill
                If you take damage from falling you fall \Prone.\\
                \D{First round 500 \Feet; Thereafter 1,500 \Feet}
            \end{entry}
            \begin{entry}{Falling on a Creature}{%
                name width=\turnLength,}
                Other creature makes a \Reflex[][val=15] and take damage equal to \CS{no damage} \SU{one-quarter} \F{half} \CF{same} you took.
            \end{entry}
            \begin{entry}{Falling Objects}{%
                name width=\turnLength,}
                Same rules as for creature. \hfill
                \D{Hazards and spells have their own rules.}
            \end{entry}
        \end{Table}\TableSpace
        \begin{Table}{Terrain}[][\pghh{423}]
            \begin{entry}{Difficult Terrain}{%
                align=center seam,
                name width=\turnLength,}
                Each square costs \Numb{5} extra \Feet of movement. \hfill
                {\scalefont{1.2}\textbf{\MSC{Greater Difficult Terrain}}}: Each square costs \Numb{10} extra \Feet. \\\hfill
                \D{If you ignore difficult terrain you move through greater difficult terrain as normal difficult terrain.}
            \end{entry}
            \begin{entry}{Hazardous Terrain}{%
                name width=\turnLength,}
                Moving through hazardous terrain deals damage.
            \end{entry}
            \begin{entry}{Narrow Surface}{%
                name width=\turnLength,}
                \OffGuard and must \Refer{Balance} to cross. When you're hit or fail a save, succeed at a \ReflexT save or \FC{fall}.
            \end{entry}
            \begin{entry}{Uneven Ground}{%
                name width=\turnLength,}
                \OffGuard and might need to \Refer{Balance} or fall \Prone. When you're hit or fail a save, succeed at a \ReflexT save or \FC{fall}.
            \end{entry}
            \begin{entry}{Incline}{%
                name width=\turnLength,}
                You need to \Refer{Climb} to ascend an incline. You're \OffGuard while \Refer{Climbing}.
            \end{entry}

        \begin{Table}{Afflictions}[][\pghh{430 - 431}]
            \begin{entry}{Iniital Exposure}{%
                name width=\turnLength,%
            }
                When first exposed you roll a saving throw. \F{Go to stage 1} \CF{Go to stage 2} \hfill
                \D{Spells with affliction typically use the caster's spell \DC}
            \end{entry}
            \begin{entry}{Stages}{%
                align=center seam,
                name width=\turnLength,}
                Each stage lasts for a time. At the end of that time you attempt a new saving.
                Move to stage \CSC{\Numb{+2}},\SC{\Numb{+1}},\FC{\Numb {-1}},\CFC{\Numb{-2}}.
                If you reach stage \Numb{0} the affliction ends.\\
                If it is virulent you must succeed at two consecutive saves to reduce the stage by \Numb{1}.
                A \CSC{CS} reduces the stage by \Numb{1}
            \end{entry}
            \begin{entry}{Multiple Exposures}{%
                name width=\turnLength,}
                If you are exposed again, make a new initial saving throw which instead
                increases the stage by \FC{\Numb{1}} \CFC{\Numb{(2)}}.
            \end{entry}
            \begin{entry}{}{%
                align=center seam,
                action={\BodyFont\scalefont{1.2}\stackbox[c][c]{\MSCB{Damage}\,\textbf{\scalefont{0.6}AND}\\\vspace{
                    -0.6ex}\MSCB{Conditions}}},
                name width=\turnLength,%
                name scale=0.95,
            }
                Any damage happens when you reach that stage.
                Conditions affect you when the you reach the stage and last for their normal duration.\\
                Any condition that doesn't have a default duration lasts as long as you're at that stage (unless stated otherwise).
            \end{entry}
